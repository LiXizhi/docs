```lua
createNode("object1",'#696969',true)
-- 轮子横梁
cylinder("union",0.7,10,'#ff0000')
rotate('z',90)

createNode("object2",'#000000',true)
-- 轮胎
cylinder("union",3,0.6,'#ff0000')
cylinder("difference",2.5,0.6,'#ff0000')
rotateNodeFromPivot("object2",'x',90,0,0,0)

createNode("object3",'#af5700',true)
-- 轮毂
for i=0, 2 do
  cylinder("union",0.25,5.2,'#ff0000')
  rotate('z',((i) * (120)))
end

createNode("object4",'#ff0000',false)
rotate('y',90)
move(4.2,0,0)
-- 合成右轮子
cloneNodeByName("union","object2",'#000000')
cloneNodeByName("union","object3",'#af4700')

createNode("object5",'#ff0000',false)
-- 左轮子
mirrorNode("object4",'yz',0,0,0)
deleteNode("object2")
deleteNode("object3")

createNode("object6",'#4e4e4e',true)
move(0,2,0)
-- 炮管
cone("union",0.8,1.2,14,'#ff0000')
cone("difference",0.6,1,11,'#ff0000')
move(0,2,0)
sphere("union",1.2,'#ff0000')
move(0,(-7),0)
sphere("union",0.6,'#ff0000')
move(0,3,0)
rotateNode("object6",'x',80)

createNode("object8",'#676767',true)
-- 横梁支架
box("union",4,0.5,2,'#ff0000')
move(0,1,0)
cone("difference",0.8,1.3,14,'#ff0000')
rotate('x',80)
move(0,2,0)

createNode("object9",'#757575',true)
-- 托盘
box("union",3,1,10,'#ff0000')
move(0,0.08,(-4.2))
cloneNodeByName("difference","object6",'#ff0000')
move(0,0,(-0.1))

createNode("object10",'#7d7d7d',true)
-- 右后支架轴
cylinder("union",0.3,6,'#ff0000')
rotate('z',90)
move(0,0,(-8.5))
prism("union",6,0.5,0.4,'#ff0000')
rotate('z',90)
move(2.7,0,(-8.5))
prism("union",6,0.5,0.4,'#ff0000')
rotate('z',90)
move(1.7,0,(-8.5))

createNode("object11",'#000000',true)
-- 右后支架
box("union",0.4,1,7,'#ff0000')
move(2.2,0,(-8.5))
rotate('x',90)
box("difference",8,0,25,'#ff0000')
move(0,1,0)

createNode("object12",'#7d7d7d',false)
-- 左后支架轴
mirrorNode("object10",'yz',0,0,0)
-- 左后支架
mirrorNode("object11",'yz',0,0,0)

createNode("object13",'#7d7d7d',true)
-- 底支架
box("union",3,1,5,'#ff0000')
move(0,(-3),(-10))
cloneNodeByName("union","object10",'#ff0000')
move(0,(-3),0)
mirrorNode("object10",'yz',0,0,0)
move(0,(-3),0)

createNode("object14",'#7d7d7d',true)
move(2,1.5,(-1))
-- 右挡板
box("union",0.5,2.5,6,'#ff0000')

createNode("object15",'#7d7d7d',true)
-- 左挡板
mirrorNode("object14",'yz',0,0,0)

createNode("object16",'#af5400',true)
-- 导火线
cylinder("union",0.1,3,'#ff0000')
move(0.1,1.2,(-6.8))
rotate('z',20)
